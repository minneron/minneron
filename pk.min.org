  {$mode delphi}{$i xpc.inc}{$H+}
  program min;
  uses xpc, mn, cw, cx, cli, kvm;
  
  var ed : mn.editor;
  begin
    kvm.ClrScr;
    ed := editor.Create;
    if ParamCount = 0 then
      writeln( 'usage : min <filename> ')
    else if ed.Load( ParamStr( 1 )) then
      begin
        ed.Run;
        ed.Destroy;
        cw.cwriteln( '|w|!k' );
        kvm.ClrScr;
      end
    else writeln( 'unable to load file: ', paramstr( 1 ))
  end.
#+end_src
