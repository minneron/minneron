program min;
  uses mn, crt, cw;
  var ed : editor;
begin
  crt.clrscr;
  ed := editor.create;
  if paramcount = 0 then
    writeln( 'usage : min <filename>' )
  else if ed.load( paramstr( 1 )) then
  begin
    ed.run;
    ed.destroy;
    cw.cwriteln( '|w|!k' );
    crt.clrscr;
  end
  else writeln( 'unable to load file: ', paramstr( 1 ));
end.
